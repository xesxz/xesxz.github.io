类声明

class PersonClass {
    constructor(name){
        this.name = name
    }
    sayNames(){

    }
}
let person = new PersonClass('nichol')
person.sayNames()

var PersonClass = class{
    constructor(name){

    }

    }
}

person.sayNames()
